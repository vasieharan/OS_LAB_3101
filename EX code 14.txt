#include &lt;stdlib.h&gt;
#define MAX_BLOCKS 50
int memory[MAX_BLOCKS] = {0}; // 0 means free, 1 means allocated
// Structure for Indexed and Linked Allocation
typedef struct {
int indexBlock;
int blocks[MAX_BLOCKS];
int blockCount;
} IndexedFile;
typedef struct {
int startBlock;
int blocks[MAX_BLOCKS];
int blockCount;
} LinkedFile;

// Sequential Allocation
void sequentialAllocation(int start, int size) {
int i, freeCount = 0;
// Check if sufficient space is available
for (i = start; i &lt; start + size; i++) {
if (memory[i] == 0) freeCount++;
}
if (freeCount == size) {
for (i = start; i &lt; start + size; i++) memory[i] = 1;
printf(&quot;File allocated from block %d to %d.\n&quot;, start, start + size - 1);
} else {
printf(&quot;Not enough contiguous free space available!\n&quot;);
}
}
// Indexed Allocation
void indexedAllocation(IndexedFile *file) {
printf(&quot;Enter index block: &quot;);
scanf(&quot;%d&quot;, &amp;file-&gt;indexBlock);
if (memory[file-&gt;indexBlock] == 1) {
printf(&quot;Index block already allocated!\n&quot;);
return;
}
printf(&quot;Enter number of blocks: &quot;);
scanf(&quot;%d&quot;, &amp;file-&gt;blockCount);
int count = 0;
for (int i = 0; i &lt; MAX_BLOCKS &amp;&amp; count &lt; file-&gt;blockCount; i++) {
if (memory[i] == 0) {
file-&gt;blocks[count++] = i;
memory[i] = 1;
}
}
if (count &lt; file-&gt;blockCount) {
printf(&quot;Not enough free blocks available!\n&quot;);
} else {
memory[file-&gt;indexBlock] = 1;
printf(&quot;File allocated with index block %d. Blocks: &quot;, file-&gt;indexBlock);
for (int i = 0; i &lt; file-&gt;blockCount; i++) printf(&quot;%d &quot;, file-&gt;blocks[i]);
printf(&quot;\n&quot;);
}
}
